add_action('init', function() {
    // Rimuove i filtri predefiniti che potrebbero interferire
    remove_filter('rest_pre_serve_request', 'rest_send_cors_headers');
    
    // Aggiunge gli header CORS personalizzati
    add_filter('rest_pre_serve_request', function($value) {
        header('Access-Control-Allow-Origin: *');
        header('Access-Control-Allow-Methods: GET, POST, OPTIONS, PUT, DELETE');
        header('Access-Control-Allow-Headers: Origin, X-Requested-With, Content-Type, Accept, Authorization, x-client-info, apikey');
        header('Access-Control-Allow-Credentials: true');
        return $value;
    });
}, 15);

// Gestione specifica per le richieste OPTIONS (Preflight)
add_filter('determine_current_user', function($user) {
    if (isset($_SERVER['REQUEST_METHOD']) && $_SERVER['REQUEST_METHOD'] == 'OPTIONS') {
        header('Access-Control-Allow-Origin: *');
        header('Access-Control-Allow-Methods: GET, POST, OPTIONS, PUT, DELETE');
        header('Access-Control-Allow-Headers: Origin, X-Requested-With, Content-Type, Accept, Authorization');
        status_header(200);
        exit;
    }
    return $user;
}, 10);
