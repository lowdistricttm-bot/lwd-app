/**
 * LOW DISTRICT - API BRIDGE
 * Crea un endpoint semplificato per l'App che non attiva i blocchi CORS standard.
 */

add_action('rest_api_init', function () {
    register_rest_route('lowdistrict/v1', '/activity', array(
        'methods' => 'GET',
        'callback' => function($request) {
            // Recupera il token JWT dalla richiesta
            $token = $request->get_param('JWT');
            
            // Simula l'autenticazione se il token è presente
            if (!empty($token)) {
                $_GET['JWT'] = $token;
            }

            // Chiama internamente BuddyPress e restituisce i dati
            $bp_request = new WP_REST_Request('GET', '/buddypress/v1/activity');
            $bp_request->set_query_params($request->get_query_params());
            $response = rest_do_request($bp_request);
            
            return rest_ensure_response($response->get_data());
        },
        'permission_callback' => '__return_true',
    ));
});

// Forza gli header CORS per il nuovo endpoint
add_filter('rest_pre_serve_request', function($value) {
    header('Access-Control-Allow-Origin: *');
    header('Access-Control-Allow-Methods: GET, POST, OPTIONS');
    header('Access-Control-Allow-Headers: Authorization, Content-Type, JWT');
    return $value;
}, 11);
