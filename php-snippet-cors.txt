/**
 * LOW DISTRICT - UNIVERSAL CORS & AUTH FIX
 * Questo codice forza i permessi di accesso per l'App e gestisce l'autenticazione JWT.
 */

// 1. Forza gli Header CORS all'inizio del caricamento
function ld_force_cors_headers() {
    // Permetti l'accesso da qualsiasi origine (necessario per l'App)
    header("Access-Control-Allow-Origin: *");
    header("Access-Control-Allow-Methods: GET, POST, OPTIONS, PUT, DELETE");
    header("Access-Control-Allow-Headers: Authorization, Content-Type, JWT, X-WP-Nonce, X-Requested-With, Accept, Origin");
    header("Access-Control-Expose-Headers: X-WP-Total, X-WP-TotalPages");

    // Rispondi immediatamente alle richieste di controllo (OPTIONS)
    if (isset($_SERVER['REQUEST_METHOD']) && $_SERVER['REQUEST_METHOD'] == 'OPTIONS') {
        status_header(200);
        exit;
    }
}
add_action('init', 'ld_force_cors_headers', 1);
add_action('rest_api_init', 'ld_force_cors_headers', 15);

// 2. Assicura che WordPress riconosca il token JWT anche se il server lo blocca negli header
add_filter('determine_current_user', function($user_id) {
    if ($user_id) return $user_id;

    $token = '';
    // Cerca il token in tutti i posti possibili
    if (isset($_GET['JWT'])) {
        $token = $_GET['JWT'];
    } elseif (isset($_SERVER['HTTP_AUTHORIZATION'])) {
        $token = str_replace('Bearer ', '', $_SERVER['HTTP_AUTHORIZATION']);
    } elseif (isset($_SERVER['HTTP_X_AUTHORIZATION'])) {
        $token = str_replace('Bearer ', '', $_SERVER['HTTP_X_AUTHORIZATION']);
    }

    if (!empty($token)) {
        // Comunica al plugin Simple JWT Login che abbiamo trovato il token
        $_GET['JWT'] = $token;
    }

    return $user_id;
}, 20);

// 3. Rimuovi i filtri standard di WordPress che potrebbero interferire
remove_filter('rest_pre_serve_request', 'rest_send_cors_headers');
